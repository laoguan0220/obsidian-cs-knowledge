#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int tree[2004];
    int lowbit(int i) {
        return i & (-i);
    }

    int sum(int x) {
        int res = 0;
        for (int i = x; i > 0; i -= lowbit(i))
            res += tree[i];
        return res;
    }

    void add(int x, int v) {
        for (int i = x; i <= n; i += lowbit(i))
            tree[i] += v;
    }

    int n;
    int end_[27];    // 改名为 end_，避免关键字
    int pre[2004];
    int arr[2004];
    int tree1[2004];

    void add1(int x, int v) {
        for (int i = x; i <= n; i += lowbit(i))
            tree1[i] += v;
    }

    int sum1(int x) {
        int res = 0;
        for (int i = x; i > 0; i -= lowbit(i))
            res += tree1[i];
        return res;
    }

    void push(int i, int temp) {
        pre[temp] = end_[i];
        end_[i] = temp;
    }

    int pop1(int i) {
        int ans = end_[i];
        end_[i] = pre[end_[i]];
        return ans;
    }

    int minMovesToMakePalindrome(string s) {
        n = s.length();

        memset(tree, 0, sizeof(tree));
        memset(tree1, 0, sizeof(tree1));
        memset(pre, 0, sizeof(pre));
        memset(end_, -1, sizeof(end_)); // ✅ 必须 -1
        memset(arr, 0, sizeof(arr));

        for (int i = 1; i <= n; i++)
            add(i, 1);

        for (int i = 0, j = 1; i < s.size(); i++, j++)
            push(s[i] - 'a', j);

        for (int i = 0, l = 1, r; i < s.size(); i++, l++) {
            if (arr[l] == 0) {
                r = pop1(s[i] - 'a');
                if (l < r) {
                    int k = sum(l);
                    arr[l] = k;
                    arr[r] = n - k + 1;
                    add(r, -1); // ✅ 放这里才对！
                } else {
                    arr[l] = (n + 1) / 2;
                }
            }
        }

        int res = 0;
        for (int i = 1; i <= n; i++) {
            res += i - 1 - sum1(arr[i]);
            add1(arr[i], 1);
        }
        return res;
    }
};

// ==================== Obsidian 运行入口 ====================
int main() {
    Solution sol;
    string s;
    cout << "请输入字符串：";
    cin >> s;
    cout << "最小交换次数 = " << sol.minMovesToMakePalindrome(s) << endl;
    return 0;
}

---
1.建立pre数组，pre[i] 有点像一个指针，指向的是字符类型与当前位置相同的前一个字符
end[i] 也像一个指针 指向的是当前位置的字符最右边的那个位置
2.建立arr数组 arr[i]的含义是第i个位置的字符最终应该到达的位置在哪里，然后最终的答案的求解利用逆序对去求解
3.在建立arr数组的过程中，如果它已经被分配过了，那么它这个位置你就不用管了，如果没有被分配，那么去求它的最右边的字符的位置，然后看当前位置的字符和最右边的字符的位置是否一样，如果一样，直接把它放在最中间 ，如果不一样，则把他们放在对称的位置，应该放的位置由树状数组去进行维护，不管哪种情况，都是要对r位置进行减去1，因为你不知道这个r位置是不是在左半平面
4.关于为什么能够想到根据arr[l]是否为0去分配情况这件事情 因为遍历的过程的目的是对数组的位置进行分配，从目的的角度去看，比较显然？？
