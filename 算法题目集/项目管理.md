nlogn的解法 二分 要求单调 约束条件下最大化某个值 而且比最终答案小的值也可以满足约束条件
可以将整体按照职级进行分类 类与类之间按照升序处理 类内部之间安装降序处理
遍历到某个职级的时候 left1 表示已经选了多少比它小的数，注意这里的数肯定能保证整体能选k个
我们可以这样想，本来这个答案我们是不知道每个类别选多少个，所以我们用二分方法，把这个答案给确定下来，同时贪心先选小的符合要求的，这样对大的没有影响
如果当前的种类能直接选够k个那么就直接选就可以了，如果不行，那么选合适的并进入到下一种类
注意这里类内部要倒序遍历 比如1 5  1 5 这里第二个1 5对第一个1 5会增加他的能够表示的数值的范围！！所以我们在写的时候一定要把每个小组块的功能阐释清楚了，需要知道自己在做什么！

# include<bits/stdc++.h>
using namespace std;
typedef struct{
    int a;
    int b;
    int id;
}persons;
const int N=2e5+5;
persons person[N];
int n;
int tok;
bool cmp(persons x,persons y)
{
    if(x.a!=y.a)
    {
        return x.a<y.a;
    }
    else
    {
        return x.b>y.b;
    }
}
vector<int>res;
int kinds[N];//每种种类的个数 肯定>=1
bool check(int k)
{
    int left1=0;
    int cur=1;
    res.clear();
    for(int i=1;i<=tok;i++)
    {
        for(int j=cur+min(kinds[i]-1,k-left1-1);j>=cur;j--)
        {
            int nowb=person[j].b;
            if(left1+j-cur+1+nowb>=k)
            {
                left1=left1+j-cur+1;
                for(int u=j;u>=cur;u--)
                {
                    res.push_back(person[u].id);
                }
                break;
            }
        }
        cur=cur+kinds[i];
    }
    if(left1>=k)
    return 1;
    else
    return 0;
}
int main()
{

//kinds数组表示的是这种种类有多少数

        for(int i=1;i<=n;i++)
        {
            if(person[i].a!=person[i-1].a)
            {
                kinds[++tok]=1;
            }

        int ans=0;
        while(l<=r)

            int midd=(l+r)/2;
